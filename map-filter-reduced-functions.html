import math

# finding the area of a circle with radius, r
def area(r):
    return math.pi * (r**2)

# if one needs to find many circle areas with the radii below
radii = [2, 7, 10]

# Without the map function
areas = []
for r in radii:
    a = area(r)
    areas.append(a)
